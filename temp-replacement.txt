async getUser(id: number): Promise<User | undefined> {
  try {
    const userDoc = await db.collection('users').doc(id.toString()).get();
    if (userDoc.exists) {
      return { id: userDoc.id, ...userDoc.data() } as User;
    }
    return undefined;
  } catch (error) {
    console.error('Error getting user:', error);
    return undefined;
  }
}
